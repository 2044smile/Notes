# 범위와 반복문
## 특정한 범위 내부의 정수들을 나열하는 자료형
```python
range(A)
# 0부터 A까지의 정수를 범위로 나열하고, A까지의 니까 A는 포함하지 않는다.

range(A, B)
# A부터 B까지의 정수를 범위로 나열

range(A, B, C)
# A부터 B까지의 정수를 범위로 나열, C 만큼씩 건너뛰면서 범위를 생성

list(range(5))
# [0,1,2,3,4]

list(range(10, 15))
# [10,11,12,13,14]

list(range(0, 10, 2))
# [0,2,4,6,8]


# **강조 +1
for i in range(0, 10 + 1):
    print(f"{i}번 째 입니다.")

for i in range(10, -1, -1): # 원래 for 0 1 2 3 4, 반대로 반복하려면 -1 을 붙이면 된다. 
    print(i)
# 10 9 8 

# **중요
for _ in range(10): # _ 식별자 입니다.
    print("반복입니다")
```
